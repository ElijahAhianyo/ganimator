def load_multiple_dataset(prefix, name_list, **kargs):
    with open(name_list, 'r') as f:
        names = [line.strip() for line in f.readlines()]
    datasets = []
    for f in names:
        kargs['filename'] = pjoin(prefix, f)
        datasets.append(MotionData(**kargs))
    return datasets
